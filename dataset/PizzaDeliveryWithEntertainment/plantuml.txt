@startuml

class User {
    Int UserID
    Date Birthdate
    String Name
    String Address
}

class Customer {
    String DeliveryAddress
}

class Order {
    Int OrderID
    DateTime OrderPlacement
    DateTime LatestDelivery
    Int NumberOfPeople
}


class EntertainmentOrder {
    String EntertainmentType
    Int Duration
}


class BusinessOwner {
    String LinkedInAccout
}

class PizzaRestaurant {
    Int ZipCode
    String Address
    String PhoneNumber
    String Website
    String OpeningHours
}

class Pizza {
    String Name
    String CrustStructure
    Double Price
}


class Entertainer {
    String StageName
    String Bio
    Double PriceThirtyMin
}

class Employment {
    String Availability
}

Customer -|> User
